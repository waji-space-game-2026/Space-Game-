    <script>
        // --- Sound Engine ---
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        function playSound(freq, duration) {
            if (audioCtx.state === 'suspended') audioCtx.resume();
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain); gain.connect(audioCtx.destination);
            osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + duration);
            osc.start(); osc.stop(audioCtx.currentTime + duration);
        }

        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        let gameState = 'MENU';
        let score = 0;
        let health = 100;
        let player = { x: 0, y: 0 };
        let stars = [];

        function init() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
            player.x = canvas.width / 2;
            player.y = canvas.height * 0.8;
            stars = Array.from({length: 80}, () => ({
                x: Math.random() * canvas.width,
                y: Math.random() * canvas.height,
                v: Math.random() * 2 + 1
            }));
        }

        function startGame() {
            score = 0; health = 100;
            gameState = 'PLAYING';
            document.getElementById('start-screen').classList.add('hidden');
            document.getElementById('over-screen').classList.add('hidden');
            document.getElementById('revive-button').style.display = 'block';
        }

        function watchAdToRevive() {
            // Earning Logic: Simulation of a Rewarded Ad
            document.getElementById('revive-button').style.display = 'none';
            document.getElementById('ad-loading').style.display = 'block';
            
            // Yahan aap apna real AdSense/AdMob code call karte hain
            setTimeout(() => {
                document.getElementById('ad-loading').style.display = 'none';
                health = 100;
                gameState = 'PLAYING';
                document.getElementById('over-screen').classList.add('hidden');
                playSound(600, 0.5);
            }, 3000); // 3 seconds simulation
        }

        function endGame() {
            gameState = 'OVER';
            document.getElementById('final-score').textContent = Math.floor(score);
            document.getElementById('over-screen').classList.remove('hidden');
        }

        function togglePause() {
            if (gameState === 'PLAYING') gameState = 'PAUSED';
            else if (gameState === 'PAUSED') gameState = 'PLAYING';
        }

        function loop() {
            ctx.fillStyle = '#010409';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            stars.forEach(s => {
                s.y += s.v; if(s.y > canvas.height) s.y = 0;
                ctx.fillStyle = 'white'; ctx.fillRect(s.x, s.y, 2, 2);
            });

            if(gameState === 'PLAYING') {
                score += 0.2;
                document.getElementById('score-val').textContent = Math.floor(score);
                document.getElementById('health-bar-fill').style.width = health + '%';
                
                // Draw Ship (Simplified for performance)
                ctx.fillStyle = '#3b82f6';
                ctx.beginPath();
                ctx.moveTo(player.x, player.y - 20);
                ctx.lineTo(player.x - 20, player.y + 20);
                ctx.lineTo(player.x + 20, player.y + 20);
                ctx.fill();

                if(Math.random() < 0.01) health -= 2;
                if(health <= 0) endGame();
            }
            requestAnimationFrame(loop);
        }

        window.onload = () => { init(); loop(); };
        window.onresize = init;

        // Mouse/Touch Control
        const handleMove = (e) => {
            if(gameState === 'PLAYING') {
                player.x = e.touches ? e.touches[0].clientX : e.clientX;
                player.y = e.touches ? e.touches[0].clientY : e.clientY;
            }
        };
        window.addEventListener('mousemove', handleMove);
        window.addEventListener('touchmove', (e) => { handleMove(e); e.preventDefault(); }, {passive: false});
    </script>
